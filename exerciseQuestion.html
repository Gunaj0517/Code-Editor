1. Find the largest element in an array of integers.


#include <iostream>
using namespace std;

int main() {
    int arr[] = {12, 35, 7, 42, 19};
    int n = sizeof(arr) / sizeof(arr[0]);
    int max = arr[0];

    for (int i = 1; i < n; i++) {
        if (arr[i] > max) {
            max = arr[i];
        }
    }

    cout << "Largest element is: " << max << endl;
    return 0;
}
        

class Test 
{ 
    static int arr[] = {12, 35, 7, 42, 19}; 
    static int largest() 
    { 
        int i; 
        int max = arr[0]; 
        for (i = 1; i < arr.length; i++) 
            if (arr[i] > max) 
                max = arr[i]; 
        
        return max; 
    } 
    public static void main(String[] args) 
    { 
        System.out.println("Largest in given array is " + largest()); 
    } 
}             
        

arr = [12, 35, 7, 42, 19]
max_value = max(arr)
print("Largest element is:", max_value)
